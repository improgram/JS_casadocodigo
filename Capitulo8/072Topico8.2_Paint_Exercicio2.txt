<script>
var tela = document.getElementById("tela");
var c = tela.getContext("2d");
c.fillStyle = "silver";
c.fillRect(0, 20, 600, 350);

function desenhaCirculo(evento){
    var x = evento.pageX;
    var y = evento.pageY;
    c.fillStyle = "red";
    c.beginPath();
    c.arc(x, y, 10, 0, 2*3.14);
    c.fill();
    alert('Posicao do Clique : ' + x + " , " + y);
};

tela.onmousemove = desenhaCirculo;
