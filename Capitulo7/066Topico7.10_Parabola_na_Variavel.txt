<script>
function mostra(frase){
    document.write(frase + '<br>')
}

var tela = document.getElementById("tela");
var c = tela.getContext("2d");

c.fillStyle="blue";

//Organizar codigo criando variavel com o FOR dentro da funcao.
function desenhaParabola() {
    for(var x = 0; x < 600; x ++){
        var y = 0.01 * x * x;
    mostra("Para X = " + x + " , o Y vale " + y);

    c.beginPath();
    c.arc(x, 400 - y, 2, 0, 2 * Math.PI);
    c.fill();    
    }
};

desenhaParabola();
