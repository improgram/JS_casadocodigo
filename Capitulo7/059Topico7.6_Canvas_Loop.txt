<script>
function verde (x, y) {
    var tela = document.getElementById("tela");
    var c = tela.getContext("2d");

    c.fillStyle="green";
    c.fillRect(x, y, 50, 50);

    c.strokeStyle="black";
    c.strokeRect(x, y, 50, 50);
};

function bolaAzul (x, y) {
    var tela = document.getElementById("tela");
    var d = tela.getContext("2d");

    d.fillStyle = "blue";
    d.beginPath();
    d.arc(50, 125, 25, 0, 2*3.14)
// X, Y , raio, Angulo inicial, Angulo final.
// De 0 ate duas vezes 3.14(Valor de PI=Circuferencia inteira)   
    d.fill();
};

var Wtest = 0;
while(Wtest < 500) {
    verde(Wtest, 0);
    Wtest = Wtest + 50;
};

for(var x = 0; x < 400; x = x + 50) {
    verde(x, 70);
    bolaAzul(x, 50);
};
